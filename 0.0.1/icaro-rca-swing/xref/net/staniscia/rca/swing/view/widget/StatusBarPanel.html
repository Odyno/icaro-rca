70  	public void addZone(String id, JComponent jc, String constraints) {
71  
72  		if (jc.getBorder() == null || jc.getBorder() instanceof UIResource) {
73  			if (jc instanceof JLabel) {
74  				jc.setBorder(new CompoundBorder(zoneBorder, new EmptyBorder(0,

83  		removeZone(zone.getId());
84  		idToZones.addFirst(zone);
85  		resync();

101 	private void resync() {
102 		removeAll();
103 		for (Iterator<JC2tStatusBarZone> iterator = idToZones.iterator(); iterator.hasNext();) {
104 			JC2tStatusBarZone type = iterator.next();
105 			add(type.getPanel(), type.getConstraint());
106 		}
107 		revalidate();
108 	}

139 /**
140  * PercentLayout. <BR>
141  * Constraint based layout which allow the space to be splitted using
142  * percentages. The following are allowed when adding components to container:
143  * <ul>
144  * <li>container.add(component); <br>
145  * in this case, the component will be sized to its preferred size
146  * <li>container.add(component, "100"); <br>
147  * in this case, the component will have a width (or height) of 100
148  * <li>container.add(component, "25%"); <br>
149  * in this case, the component will have a width (or height) of 25 % of the
150  * container width (or height) <br>
151  * <li>container.add(component, "*"); <br>
152  * in this case, the component will take the remaining space. if several
153  * components use the "*" constraint the space will be divided among the
154  * components.
155  * </ul>
156  * 
157  * @javabean.class name="PercentLayout"
158  *                 shortDescription="A layout supports constraints expressed in percent."
159  */
160 class PercentLayout implements LayoutManager2 {

258 	public void setConstraint(Component component, Object constraints) {
259 		if (constraints instanceof Constraint) {
260 			m_ComponentToConstraint.put(component, constraints);
261 		} else if (constraints instanceof Number) {
262 			setConstraint(component, new NumberConstraint(
263 					((Number) constraints).intValue()));
264 		} else if ("*".equals(constraints)) {
265 			setConstraint(component, REMAINING_SPACE);
266 		} else if ("".equals(constraints)) {
267 			setConstraint(component, PREFERRED_SIZE);
268 		} else if (constraints instanceof String) {
269 			String s = (String) constraints;
270 			if (s.endsWith("%")) {
271 				float value = Float.valueOf(s.substring(0, s.length() - 1))
272 						.floatValue() / 100;
273 				if (value > 1 || value < 0)
274 					throw new IllegalArgumentException(
275 							"percent value must be >= 0 and <= 100");
276 				setConstraint(component, new PercentConstraint(value));
277 			} else {
278 				setConstraint(component,
279 						new NumberConstraint(Integer.valueOf(s)));
280 			}
281 		} else if (constraints == null) {
282 			// null constraint means preferred size
283 			setConstraint(component, PREFERRED_SIZE);
284 		} else {
285 			throw new IllegalArgumentException("Invalid Constraint");
286 		}
287 	}

366 	public Dimension preferredLayoutSize(Container parent) {
367 		Component[] components = parent.getComponents();
368 		Insets insets = parent.getInsets();
369 		int width = 0;
370 		int height = 0;
371 		Dimension componentPreferredSize;
372 		boolean firstVisibleComponent = true;
373 		for (int i = 0, c = components.length; i < c; i++) {
374 			if (components[i].isVisible()) {
375 				componentPreferredSize = components[i].getPreferredSize();
376 				if (orientation == HORIZONTAL) {
377 					height = Math.max(height, componentPreferredSize.height);
378 					width += componentPreferredSize.width;
379 					if (firstVisibleComponent) {
380 						firstVisibleComponent = false;
381 					} else {
382 						width += gap;
383 					}
384 				} else {
385 					height += componentPreferredSize.height;
386 					width = Math.max(width, componentPreferredSize.width);
387 					if (firstVisibleComponent) {
388 						firstVisibleComponent = false;
389 					} else {
390 						height += gap;
391 					}
392 				}
393 			}
394 		}
395 		return new Dimension(width + insets.right + insets.left, height
396 				+ insets.top + insets.bottom);
397 	}

399 	public void layoutContainer(Container parent) {
400 		Insets insets = parent.getInsets();
401 		Dimension d = parent.getSize();
402 
403 		// calculate the available sizes
404 		d.width = d.width - insets.left - insets.right;
405 		d.height = d.height - insets.top - insets.bottom;
406 
407 		// pre-calculate the size of each components
408 		Component[] components = parent.getComponents();
409 		int[] sizes = new int[components.length];
410 
411 		// calculate the available size
412 		int totalSize = (HORIZONTAL == orientation ? d.width : d.height)
413 				- (components.length - 1) * gap;
414 		int availableSize = totalSize;
415 
416 		// PENDING(fred): the following code iterates 4 times on the component
417 		// array, need to find something more efficient!
418 
419 		// give priority to components who want to use their preferred size or
420 		// who
421 		// have a predefined size
422 		for (int i = 0, c = components.length; i < c; i++) {
423 			if (components[i].isVisible()) {
424 				Constraint constraint = (Constraint) m_ComponentToConstraint
425 						.get(components[i]);
426 				if (constraint == null || constraint == PREFERRED_SIZE) {
427 					sizes[i] = (HORIZONTAL == orientation ? components[i]
428 							.getPreferredSize().width : components[i]
429 							.getPreferredSize().height);
430 					availableSize -= sizes[i];
431 				} else if (constraint instanceof NumberConstraint) {
432 					sizes[i] = ((NumberConstraint) constraint).intValue();
433 					availableSize -= sizes[i];
434 				}
435 			}
436 		}
437 
438 		// then work with the components who want a percentage of the remaining
439 		// space
440 		int remainingSize = availableSize;
441 		for (int i = 0, c = components.length; i < c; i++) {
442 			if (components[i].isVisible()) {
443 				Constraint constraint = (Constraint) m_ComponentToConstraint
444 						.get(components[i]);
445 				if (constraint instanceof PercentConstraint) {
446 					sizes[i] = (int) (remainingSize * ((PercentConstraint) constraint)
447 							.floatValue());
448 					availableSize -= sizes[i];
449 				}
450 			}
451 		}
452 
453 		// finally share the remaining space between the other components
454 		ArrayList remaining = new ArrayList();
455 		for (int i = 0, c = components.length; i < c; i++) {
456 			if (components[i].isVisible()) {
457 				Constraint constraint = (Constraint) m_ComponentToConstraint
458 						.get(components[i]);
459 				if (constraint == REMAINING_SPACE) {
460 					remaining.add(new Integer(i));
461 					sizes[i] = 0;
462 				}
463 			}
464 		}
465 
466 		if (remaining.size() > 0) {
467 			int rest = availableSize / remaining.size();
468 			for (Iterator iter = remaining.iterator(); iter.hasNext();) {
469 				sizes[((Integer) iter.next()).intValue()] = rest;
470 			}
471 		}
472 
473 		// all calculations are done, apply the sizes
474 		int currentOffset = (HORIZONTAL == orientation ? insets.left
475 				: insets.top);
476 
477 		for (int i = 0, c = components.length; i < c; i++) {
478 			if (components[i].isVisible()) {
479 				if (HORIZONTAL == orientation) {
480 					components[i].setBounds(currentOffset, insets.top,
481 							sizes[i], d.height);
482 				} else {
483 					components[i].setBounds(insets.left, currentOffset,
484 							d.width, sizes[i]);
485 				}
486 				currentOffset += gap + sizes[i];
487 			}
488 		}
489 	}
